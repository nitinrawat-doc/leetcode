class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int max_profit=0;
        int best_buy=prices[0];
        for(int i=0;i<prices.size();i++){
            if(prices[i]>best_buy){
                max_profit=max(max_profit,prices[i]-best_buy);
            }
            best_buy=min(prices[i],best_buy);
        }
        return max_profit;
    }
};
