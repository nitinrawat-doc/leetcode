class Solution {
public:
    vector<int> getSneakyNumbers(vector<int>& nums) {
        vector<int> result;
        unordered_map<int, int> f;
        for (int n : nums) {
            f[n]++;
            if(f[n]==2){
                result.push_back(n);
            }
        }
        return result;
        
    }
};
