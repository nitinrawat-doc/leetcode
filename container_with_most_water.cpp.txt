class Solution {
public:
    int maxArea(vector<int>& height) {
        int left=0;
        int right=height.size()-1;
        int maxa=0;
        while(left<right){
            int width=right-left;
            int h=min(height[left],height[right]);
            int area=width*h;
            maxa=max(maxa,area);
            if(height[left]<height[right]){
                left++;
            }
            else{
                right--;
            }
        }
        return maxa;
    }
};
