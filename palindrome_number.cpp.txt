class Solution {
public:
    bool isPalindrome(int x) {
        long long num = x;
        long long on = num;
        long long r = 0;
        while (num > 0) {
            int ld = num % 10;
            r = (r * 10) + ld;
           num= num / 10;
        }
        if (r == on)
            return true;
            else return false;
    }
};
